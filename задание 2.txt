#include <iostream>
#include <stdio.h>
using namespace std;
void main() {
	int m, n, i, elem, min;
	cout << " Vvedite razmer matricy:  ";
	cin >> n >> m;
	int ** a = new int * [n];
	for (i = 0; i < n; i++) {
		a[i] = new int[m];

		for (int j = 0; j < m; j++) {
			a[i][j] = ((rand() % 10));
		}
	}
	cout << endl << endl;
	for (int i = 0; i < n; i++) {
		for (int j = 0; j < m; j++)
			cout << a[i][j] << " ";
		cout << endl;
	}
	int ** b = new int *[n];
	b = a;
	for (i = 0; i < n; i++) {
		min = b[i][0];
		for (int j = 1; j < m; j++) {
			if (b[i][j] < min)
				min = b[i][j];
		}
		cout << "Min element stroki " << i+1 << " = " << min << endl;
	}
	cout << "\n";
	delete[] a;
	system("pause");
}
